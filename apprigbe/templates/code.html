Function that finds all the genes with in 1MB of the marker in the UCSC DB

#!/usr/bin/env python
# -*- coding: utf-8 -*-

def snp_gene(marker_id):

db = MySQLdb.connect(host="genome-mysql.cse.ucsc.edu",user="genome",db="hg19")

cur = db.cursor()

cur.execute("SELECT K.proteinID,K.name,S.name,S.chrom,S.chromStart,S.chromEnd,S.alleleFreqs FROM snp135 as S left join knownGene as K on(S.chrom=K.chrom and not(K.txEnd+1000000< S.chromStart or S.chromEnd+1000000< K.txStart)) WHERE S.name=%s ",(marker_id))

for row in cur.fetchall() :

yield row

 

A module that contains functions that compare two variants(based on their chromosomes and positions) and that check if a variant is homo/heterozygous

#!/usr/bin/env python
# -*- coding: utf-8 -*-

#function that returns true if two variants are the same and false otherwise

def check_comm_var(ch1,pos1,ch2,pos2):
if ch1==ch2 and pos1==pos2:
return True
else:
return False

#function that returns true if a variant is homozygous, false otherwise

def check_homo(c_ratio):
if 0 <= c_ratio and c_ratio <=0.2:
return True
else:
return False

#function that returns true if a variant is heterozygous or wildtype, false otherwise

def check_hetero_wt(c_ratio)
if 0.2 < c_ratio and c_ratio <=1:
return True
else:
return False
